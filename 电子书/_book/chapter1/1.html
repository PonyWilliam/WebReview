添加10个a标签的代码

<!--
    10个a标签，点击每个都提示对应的序号
-->
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
</head>
<style>
    a{
        color:red;
        display: block;
    }
</style>
<body>
    <div id="tests"></div>
</body>
<script>
    var ele = document.getElementById('tests')
    for(let i = 0;i<10;i++){
        var a = document.createElement('a')
        a.innerHTML = i + 1
        a.onclick = function(){
            alert(i + 1)
        }
        ele.append(a)
    }
</script>
</html>
